'use strict';

import {Cielo} from './cielo.js';
import {Bola} from './bola.js';
/**
 * @description Ejecuta las funciones del canvas
 * @param {Number} anchura - Anchura del canvas
 * @param {Number} altura - Altura del canvas
 */
const ejecutar = function(anchura, altura) {
  const cielo = new Cielo(anchura, altura);
  cielo.inicializar();
  const bola = new Bola();
  bola.dibujar();
  boton(cielo, bola, 'arriba');
  boton(cielo, bola, 'izquierda');
  boton(cielo, bola, 'derecha');
  boton(cielo, bola, 'abajo');
  const color = document.getElementById('Color');
  color.oninput = function() {
    bola.setColor(color.value);
    bola.dibujar();
  };
};

const boton = function(cielo, bola, botonNombre) {
  const botones = document.querySelectorAll('.' + botonNombre);
  botones.forEach((el) => {
    el.addEventListener('click', contar);
  });
  function contar() {
    const boton = document.getElementById(botonNombre);
    cielo.inicializar();
    if (!bola.movimiento(botonNombre)) {
      bola.dibujar();
      boton.style.backgroundColor = 'red';
    } else {
      boton.style.backgroundColor = 'green';
    }
    if (botonNombre === 'abajo') {
      const boton = document.getElementById('arriba');
      boton.style.backgroundColor = 'green';
    }
    if (botonNombre === 'arriba') {
      const boton = document.getElementById('abajo');
      boton.style.backgroundColor = 'green';
    }
    if (botonNombre === 'derecha') {
      const boton = document.getElementById('izquierda');
      boton.style.backgroundColor = 'green';
    }
    if (botonNombre === 'izquierda') {
      const boton = document.getElementById('derecha');
      boton.style.backgroundColor = 'green';
    }
  }
};


window.ejecutar = ejecutar;
